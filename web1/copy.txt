            if (newVol < 0) newVol = 0;
            audio.volume = newVol;

            if (currentStep >= (duration / 50)) {
                audio.volume = targetVolume;
                if (targetVolume === 0) audio.pause();
                clearInterval(interval);
            }
        }, 50);
    }

    function scrollToBottom() {
        setTimeout(() => { chatContainer.scrollTop = chatContainer.scrollHeight; }, 50); 
    }

    function addMessage(content, sender, isImage = false, isAudio = false) {
        const wrapper = document.createElement("div");
        wrapper.className = `msg-wrapper msg-${sender}`; 
        
        if (isImage) {
            const imgBubble = document.createElement("div");
            imgBubble.className = "msg-bubble-img";
            const img = document.createElement("img");
            img.src = content;
            imgBubble.appendChild(img);
            wrapper.appendChild(imgBubble);
        } else if (isAudio) {
            const audioBubble = document.createElement("div");
            audioBubble.className = "msg-bubble-audio";
            const audio = document.createElement("audio");
            audio.controls = true;
            audio.src = content;
            audioBubble.appendChild(audio);
            wrapper.appendChild(audioBubble);
        } else {
            const bubble = document.createElement("div");
            bubble.className = "msg-bubble";
            bubble.innerText = content;
            wrapper.appendChild(bubble);
        }

        chatContainer.appendChild(wrapper);
        scrollToBottom();
    }

    function clearActions() { actionArea.innerHTML = ""; }

        toggleBtn.style.pointerEvents = "none";
        reportBtn.style.pointerEvents = "none";

        elements.forEach(el => { el.disabled = true; el.style.pointerEvents = "none"; });

    function openReportModal() { reportModal.style.display = "flex"; }

        closeReportModal(); 
        disableInputs();    

        if (isGlitchActive) {
            clearInterval(glitchTimerInterval);
            glitchModal.style.display = "none";

            questionsData[15].options = [ { text: "是的，維持現狀就足夠了。", add: {} }, { text: "不，我覺得我還能夠改變。", add: {} } ];
            setTimeout(() => { renderQuestion(); enableInputs(); }, 1000);

        scores["傑克"] += 1;
        addMessage("並無檢測到問題。", "system");
        setTimeout(() => { showCurrentQuestion(); enableInputs(); }, 500);
    }

    function toggleInputMode() {
        isStickerMode = !isStickerMode;
        toggleBtn.className = isStickerMode ? "tool-btn btn-purple" : "tool-btn btn-yellow";
        renderActionContent(); 
    }

    function renderActionContent() {
        clearActions();
        if (isStickerMode) {
            const grid = document.createElement("div");
            grid.className = "sticker-grid";
            stickersList.forEach(src => {
                const img = document.createElement("img");
                img.src = src; img.className = "sticker-img";
                img.onclick = () => handleStickerAnswer(src);
                grid.appendChild(img);
            });
            actionArea.appendChild(grid);
        } else {
            const currentQ = questionsData[currentQuestionIndex];
            currentQ.options.forEach(option => {
                const btn = document.createElement("button");
                btn.className = "option-btn"; btn.innerText = option.text;
                btn.onclick = () => handleAnswer(option);
                actionArea.appendChild(btn);
            });
        }
        scrollToBottom();
    }

    function showCurrentQuestion() {
        const currentQ = questionsData[currentQuestionIndex];
        if (currentQ.audio) {
            addMessage(currentQ.audio, "system", false, true);
        }
        addMessage(currentQ.q, "system");
    }

    function renderQuestion() {
        isStickerMode = false;
        toggleBtn.className = "tool-btn btn-yellow";
        toolbar.style.display = "flex"; 
        toggleBtn.style.display = currentQuestionIndex < 2 ? "none" : "block";

        // --- Index 9 傳出時漸弱，Index 10 開始時漸強 ---
        if (currentQuestionIndex === 9) { 
            fadeAudio(bgmAudio, 0, 1000); 
        } else if (currentQuestionIndex === 10) { 
            if (bgmAudio.paused && bgmAudio.src) {
                bgmAudio.play().catch(e => console.log(e));
            }
            fadeAudio(bgmAudio, 1, 1000); 
        }

        enableInputs();
        showCurrentQuestion(); 
        renderActionContent(); 
    }

    function handleAnswer(option) {
        disableInputs();
        addMessage(option.text, "user");
        if (option.add) {
            for (let character in option.add) { scores[character] += option.add[character]; }
        }
        
        // --- 攔截跳過貼圖的動作 ---
        if (option.action === "sticker_skip") {
            setTimeout(() => {
                addMessage("好吧沒關係!那我們繼續", "system");
                triggerNextQuestion();
            }, 500);
            return;
        }

        if (option.action === "play_music") {
            bgmAudio.src = musicList[currentMusicIndex];
            bgmAudio.volume = 1;
            bgmAudio.play().catch(e => console.log(e));
            triggerNextQuestion(); 
        } else if (option.action === "skip_music") {
            setTimeout(() => { currentQuestionIndex = 2; renderQuestion(); }, 500);
        } else if (option.action === "change_music") {
            musicSwitchCount++;
            currentMusicIndex = (currentMusicIndex + 1) % musicList.length;
            bgmAudio.src = musicList[currentMusicIndex];
            bgmAudio.play().catch(e => console.log(e));
            setTimeout(() => {
                addMessage("已為您切換至下一首，滿意現在的音樂嗎？", "system");
                enableInputs(); renderActionContent(); 
            }, 500);
        } else if (option.action === "keep_music") {
             if (musicSwitchCount === 1) {
                scores["白雪"] += 1; scores["鋼琴家"] += 1; scores["愛麗絲"] += 1; scores["航海家"] += 1;
            } else if (musicSwitchCount === 2) {
                scores["鋼琴家"] += 1; scores["愛麗絲"] += 1;
            } else if (musicSwitchCount > 2 && musicSwitchCount <= 5) {
                scores["鋼琴家"] += 1; scores["愛麗絲"] += 1; scores["傑克"] += 1; scores["壁畫"] += 1;
            } else if (musicSwitchCount > 5) {
                scores["鋼琴家"] += 2; scores["愛麗絲"] += 1; scores["傑克"] += 1; scores["壁畫"] += 1;
            }
            triggerNextQuestion(); 
        } else {
            triggerNextQuestion();
        }
    }

    function handleStickerAnswer(src) {
        disableInputs();
        addMessage(src, "user", true);
        if (!stickerScoreMap[currentQuestionIndex]) { triggerNextQuestion(); return; }
        const filename = src.split('/').pop(); 
        let stickerType = 3; 
        if (filename === '1.png') stickerType = 1;
        else if (filename === '2.png') stickerType = 2;
        const currentScoreLogic = stickerScoreMap[currentQuestionIndex][stickerType];
        
        if (currentScoreLogic.optIndex !== null) {
            const baseAdd = questionsData[currentQuestionIndex].options[currentScoreLogic.optIndex].add;
            for (let character in baseAdd) { scores[character] += baseAdd[character]; }
        }
        if (currentScoreLogic.bonus) {
            for (let character in currentScoreLogic.bonus) { scores[character] += currentScoreLogic.bonus[character]; }
        }
        
        // --- 攔截貼圖教學題 (Index 2) 給予回饋 ---
        if (currentQuestionIndex === 2) {
            setTimeout(() => {
                addMessage("很有趣的選擇呢!我也很喜歡這張，那讓我們繼續吧。", "system");
                triggerNextQuestion();
            }, 500);
        } else {
            triggerNextQuestion();
        }
    }

    function triggerNextQuestion() {
        setTimeout(() => {
            currentQuestionIndex++;
            // --- Index 10 觸發抽卡環節 ---
            if (currentQuestionIndex === 10 && !isCardEventDone) {
                startCardEvent();
            } else if (currentQuestionIndex < questionsData.length) {
                renderQuestion(); 
            } else if (!isGlitchDone) {
                startGlitchEvent();
            } else {
                showResult(); 
            }
        }, 500); 
    }

    function startGlitchEvent() {
        isGlitchActive = true; closeBtnClickCount = 0; let timeLeft = 15;
        clearActions();
        const gibberishList =["!@#$*%", "?><&%^", "§¶∆∞∑", "▚▞▚▞"];
        for (let i = 0; i < 3; i++) {
            const btn = document.createElement("button");
            btn.className = "option-btn"; btn.innerText = gibberishList[Math.floor(Math.random() * gibberishList.length)];
            btn.disabled = true; actionArea.appendChild(btn);
        }
        disableInputs(); reportBtn.style.pointerEvents = "auto";
        document.getElementById("glitch-timer").innerText = timeLeft;
        glitchModal.style.display = "flex";
        glitchTimerInterval = setInterval(() => {
            timeLeft--;
            document.getElementById("glitch-timer").innerText = timeLeft;
            if (timeLeft <= 0) { glitchTimeout(); }
        }, 1000);
    }

    function handleGlitchClose() {
        closeBtnClickCount++;
        const content = document.getElementById("glitch-modal-content");
        content.classList.remove("shake"); void content.offsetWidth; content.classList.add("shake");
    }

    function glitchTimeout() {
        if (!isGlitchActive) return;
        clearInterval(glitchTimerInterval); glitchModal.style.display = "none"; closeReportModal(); 
        isGlitchActive = false; isGlitchDone = true;
        if (closeBtnClickCount >= 3) { scores["傑克"] += 3; scores["壁畫"] += 1; } 
        else { scores["白雪"] += 1; scores["愛麗絲"] += 1; scores["鋼琴家"] += 1; scores["問號頭人"] += 2; }
        addMessage("系統已自動恢復。", "system");
        setTimeout(() => { showResult(); }, 1000);
    }

            [cardPool[i], cardPool[j]] = [cardPool[j], cardPool[i]];

        toolbar.style.display = "none"; drawNextCard();
    }

    function finishCardEvent(state) {
        if (state === 'force_4') { scores["科學家"] += 4; } 
        else if (state === 'stop_at_3') { scores["科學家"] += 1; } 
        else if (drawCount === 1 && state === true) { scores["白雪"] += 2; scores["問號頭人"] += 2; } 
        else if (drawCount === 1 && state === false) { scores["航海家"] += 3; } 
        else if ((drawCount === 2 || drawCount === 3) && (state === true || state === false)) {
            scores["傑克"] += 1; scores["愛麗絲"] += 1; scores["鋼琴家"] += 1; scores["壁畫"] += 1;
        }
        setTimeout(() => { renderQuestion(); }, 500);
    }

    function showResult() {
        clearActions(); toolbar.style.display = "none"; 
        let highestScore = -1; let resultCharacter = "";
        for (let character in scores) {
            if (scores[character] > highestScore) { highestScore = scores[character]; resultCharacter = character; }
        }
        addMessage(`測驗結束！你的專屬標籤是：【${resultCharacter}】`, "system");
        let scoreDetails = Object.entries(scores).map(([char, score]) => `${char}: ${score}`).join(' / ');
        setTimeout(() => { addMessage(`[除錯資訊] ${scoreDetails}`, "system"); }, 400); 
        const restartBtn = document.createElement("button");
        restartBtn.className = "option-btn"; restartBtn.innerText = "再測一次";
        restartBtn.onclick = () => location.reload();
        actionArea.appendChild(restartBtn);
    }
